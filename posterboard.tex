      \textbf{YES/NO Securities}
      \begin{itemize}
        \item Price $\in [0,1]$ represents implied probability
        \item YES security pays \$1 if event occurs
        \item NO security pays \$1 if event does not occur
      \end{itemize}

      \textbf{Probability Space}
      \begin{itemize}
        \item $(\Omega, \mathcal{F}, P)$: sample space, events, probability measure
        \item $\Omega$: all possible states of the world
        \item Filtration $\mathbb{F} = (\mathcal{F}_t)$: information flow over time
      \end{itemize}

      \textbf{Martingale}
      \begin{itemize}
        \item Model for a fair game
        \item $E[X_{t+1} | \mathcal{F}_t] = X_t$
        \item Current price equals expected future value
      \end{itemize}

      \textbf{Brownian Motion}
      \begin{itemize}
        \item Continuous-time stochastic process
        \item $W(t) \sim N(0, t)$ with continuous paths
        \item $\sqrt{\Delta t}$ scaling for variance consistency
      \end{itemize}

    \end{column}

    \heading{Weighted Wallet Signal}

    \begin{columns}[t]
      \begin{column}{0.55\linewidth}
        \begin{itemize}
          \setlength{\itemsep}{0pt}\setlength{\parsep}{0pt}
          \item $p \in [0, 1]$ : Price in dollars
          \item $n$ : Number of wallets in current market
          \item $W_{\text{score}} \in [0, 100]$ : Wallet's insider score
          \item $\text{Pos}$ : Position size (number of shares)
          \item $dir \in \{-1, 1\}$ : direction ($-1$ short, $+1$ long)
        \end{itemize}
      \end{column}
      \begin{column}{0.45\linewidth}
        \[ Signal = \sum_{i=0}^n\frac{W_{score}\cdot\text{Pos}\cdot p\cdot dir}{W_{score}\cdot\text{Pos}\cdot p} \]
      \end{column}
    \end{columns}



    \heading{Entry \& Exit Rules}

    \begin{itemize}
      \setlength{\itemsep}{0pt}\setlength{\parsep}{0pt}
      \item $\theta_{\text{entry}}$ : Minimum signal magnitude to enter a trade 
      \item $\theta_{\text{exit}}$ : Signal magnitude threshold to exit a trade (weakened signal)
    \end{itemize}

    \begin{table}
      \centering
      \small
      \begin{tabular}{l l}
        \toprule
        \textbf{Condition} & \textbf{Rule} \\
        \midrule
        Enter & $|\text{signal}| > \theta_{\text{entry}}$, price $\in [0.10, 0.90]$ \\
        Exit---Weakened & Signal drops below $\theta_{\text{exit}}$ \\
        Exit---Reversal & Signal flips sign \\
        Exit---Extreme & Price $< 0.02$ or $> 0.99$ \\
        Force Close & Market expiry \\
        \bottomrule
      \end{tabular}
    \end{table}

    \heading{Position Sizing}
    Given the portfolio risk fraction ($\gamma$) and the consensus signal ($s$), the number of shares to purchase is:

    \[
      \text{shares} = \left\lfloor \frac{\text{cash} \times \gamma \times |s|}{\text{price}} \right\rfloor
    \]
    where $s$ is the normalized signal from the wallet consensus calculation, with magnitude $|s| \in [0, 1]$.

  \begin{block}{Evolutionary Parameter Optimization}

    The system runs \textbf{dozens of parallel strategy instances} and uses an in-process evolutionary algorithm:

    \begin{enumerate}
      \setlength{\itemsep}{0pt}\setlength{\parsep}{0pt}
      \item \textbf{Gen-1 Sweep} (53 instances, 5{,}447 trades): grid search over entry/exit thresholds, sizing modes, wallet counts, timing modes.
      \item \textbf{Selection}: top 7 performers converged on entry 0.68--0.75, exit 0.16--0.20, consensus $\geq 0.70$.
      \item \textbf{Gen-2 Evolution} (30 containers, 5 families): HC-Lean mutations, HC-Bigger mutations, crossbreeds, timing variants.
      \item \textbf{Auto-Evolver Daemon}: runs every 20 minutes, identifies underperformers (PF $< 0.5$), blends them 70\% toward PnL-weighted winning donor with 5\% entropy. 
    \end{enumerate}

  \end{block}  



  \begin{block}{Model II: Earnings Convergence Model}
    The model identifies alpha by detecting convergence between prediction market probabilities and fundamental analyst data.

    \textbf{1. Signal Normalization ($x_i$)}

    \begin{columns}[t]
      \begin{column}{0.52\linewidth}
        \begin{itemize}
          \setlength{\itemsep}{4pt}\setlength{\parsep}{0pt}
          \item \textbf{Polymarket Price ($x_1$):} $p \in [0,1]$ normalized by log liquidity $V$
          \item \textbf{SmartEstimate Surprise ($x_2$):} StarMine surprise $S_{pct}$ over 20\% threshold
          \item \textbf{Analyst Sentiment ($x_3$):} rescales recommendations $A_{rec} \in [1,5]$
        \end{itemize}
      \end{column}
      \begin{column}{0.48\linewidth}
        \[ x_1 = 2(p - 0.5) \cdot \min\!\left( \tfrac{\log_{10}(V)}{5}, 1 \right) \]
        \[ x_2 = \max\!\left(\min\!\left(\tfrac{S_{pct}}{20}, 1\right), -1\right) \]
        \[ x_3 = \frac{A_{rec} - 3}{2} \]
      \end{column}
    \end{columns}

  \end{block}

    \textbf{2. Execution Logic (Binary)}
    A trade is initiated \textbf{if and only if} all signals show total directional consensus:
    \[ \text{Condition: } \text{sgn}(x_1) = \text{sgn}(x_2) = \text{sgn}(x_3) \neq 0 \]
    That is, all three components must have the same sign (all positive or all negative). Each component is bounded such that $x_i \in [-1, 1]$.

    \textbf{3. Portfolio Allocation \& Risk Management ($\Phi$)}
    If the execution condition is met, the allocation $\Phi$ (fraction of total portfolio) is determined by scaling the signal by the \textbf{portfolio risk fraction} ($\gamma$):
    \[ \Phi = \gamma \cdot \left( \sum_{i=1}^{3} w_i x_i \right), \quad \gamma \in (0, 1] \]
where $\sum_{i=1}^{3} w_i = 1$. Currently, weights are initialized equally: $w_1 = w_2 = w_3 = \frac{1}{3}$ (optimization pending).
    The component consensus acts as the \textit{execution gate}, while the magnitude of $\Phi$ determines the \textit{capital commitment}.

  \end{block}
